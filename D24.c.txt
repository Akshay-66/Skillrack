void swapFirstHalfSecondHalf(char *str)
{
    int q = strlen(str);
    int w = q/2;
    int e = 0;
    char x[100];
    for(int i=w; i<q; i++)
    {
        x[e] = str[i];
        e++;
    }
    for(int i=0; i<w; i++)
    {
        x[e] = str[i];
        e++;
    }
    for(int i=0; i<e; i++){

int alp(char a[100], int x)
{
    int i,j,q=0;
    for(i=x; i<x+5; i++)
    {
        if(isalpha(a[i]))
        {
            q++;
        }
    }
    if(q==5)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}

int num(char a[100], int x)
{
    int i,j,q=0;
    for(i=x; i<x+5; i++)
    {
        if(isdigit(a[i]))
        {
            q++;
        }
    }
    if(q==5)
    {
        return 1;
    }
    else
    {
        return 0;
    }
}
