#include<stdlib.h>

int main()
{
    int i,j,a[100][100],r=0,n,x=0;
    scanf("%d", &n);
    for(i=0; i<n; i++)
    {
        for(j=0; j<n; j++)
        {
            scanf("%d", &a[i][j]);
        }
    }
    for(i=0; i<n; i++)
    {
        r = 0;
        for(j=0; j<n; j++)
        {
            r = r + a[j][i];
        }
        if(r==0)
        {
            x++;
        }
    }
    if(n==x)
    {
        printf("yes");
    }
    else
    {
        printf("no");
    }
}
